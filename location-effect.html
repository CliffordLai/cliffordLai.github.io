4.5 Count by Neigbourhood

# Aggregate the number of collisions by neighbours
collisionByNeighbor <- collision %>%
  drop_na(neighborhood) %>%
  group_by(neighborhood) %>%
  summarise(N=n(),Latitude=mean(lat),Longitude=mean(lon))

# Total number of collisions by neighbours
ggmap(map_london) +
  geom_point(data = collisionByNeighbor, 
             aes(x = Longitude, y = Latitude, size=N), 
             alpha = .9, color="red") + 
  geom_text(data = collisionByNeighbor, 
            aes(x = Longitude, y = Latitude, label=N), 
            size=5, color="blue",  fontface = "bold", show.legend = FALSE)+
  geom_text(data = collisionByNeighbor, 
            aes(x = Longitude, y = Latitude+0.003, label=neighborhood), 
            size=3, color="blue",  fontface = "bold", show.legend = FALSE)+
  labs(title = str_c('Number of Collisions in London Neighbourhoods\n', 
                     'of London from 2006 to 2017'
  ),
  #subtitle = 'Here is where your subtitle goes',
  size = str_c('Number of', '\ncollisions')
  ) +
  theme(text = element_text(color = "#444444"),
        plot.title = element_text(size = 18, face = 'bold'),
        plot.subtitle = element_text(size = 12),
        axis.text = element_blank(),
        axis.title = element_blank(),
        axis.ticks = element_blank()
  )
